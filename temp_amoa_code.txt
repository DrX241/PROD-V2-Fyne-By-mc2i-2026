      try {
        // Utiliser notre helper pour extraire le JSON de la réponse
        const synthesis = extractJsonFromOpenAiResponse(response);
        
        if (synthesis) {
          res.json({
            success: true,
            synthesis,
            model: openAIService.getCurrentModelName()
          });
        } else {
          // Si l'extraction du JSON a échoué, tenter de créer un JSON de secours
          const expectedFields = [
            "presentation", "parcours", "impressions", "motivations", 
            "projet", "potentiel", "processus", "criteres", 
            "forces", "faiblesses", "anglais", "stage", 
            "synthese", "raison"
          ];
          
          const fallbackJson = createFallbackJson(response, expectedFields);
          
          if (fallbackJson) {
            console.log("Utilisation d'un format JSON de secours pour les notes");
            res.json({
              success: true,
              synthesis: fallbackJson,
              model: openAIService.getCurrentModelName(),
              fallbackMode: true
            });
          } else {
            // En dernier recours, renvoyer la réponse brute
            console.error("Impossible de traiter la réponse OpenAI en JSON");
            res.status(500).json({ 
              success: false,
              error: "Erreur lors du parsing de la synthèse",
              rawResponse: response
            });
          }
        }
      } catch (parseError) {
        console.error("Erreur lors du traitement de la synthèse:", parseError);
        
        // Si une erreur inattendue se produit, renvoyer l'erreur et la réponse brute
        res.status(500).json({ 
          success: false,
          error: `Erreur lors du traitement de la synthèse: ${parseError.message}`,
          rawResponse: response
        });
      }
